	if(canje_puntos >= monto_compra){
		printf("Los puntos acumulados cubren la compra.\n");

		printf("Si decea pagar en efectivo presione 2\n");
		scanf("%i", &opcion);
		if(opcion == 1){
			puntos_acumulados = (canje_puntos/10) - (monto_compra/10);
		}
	
	}else{
		printf("Los puntos acumulados no cubren la compra.\n");
	}

	printf("Ingrese codigo de producto: ");
	scanf("%i", &codigo_producto);
		if (100<=codigo_producto && codigo_producto<200){
			if(monto_compra<10000){
				puntos_acumulados = puntos_acumulados + (monto_compra/100);
			}else{
				puntos_acumulados = puntos_acumulados + (monto_compra/50);
			}

		}else if(200<=codigo_producto && codigo_producto<300){
				if(monto_compra<10000){
					puntos_acumulados = puntos_acumulados + (monto_compra/50);
				}else{
					puntos_acumulados = puntos_acumulados + (monto_compra/30);
				}

		}else{
			printf("El codigo de producto ingresado no es valido");
		}
	
	printf("Ahora tienes %i puntos acumulados", puntos_acumulados);

	return 0;
}
